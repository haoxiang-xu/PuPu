from dataclasses import dataclass, field from typing import Optional PROMPT_SECTION_KEYS = ( "identity", "capability", "workflow", "delegation", "constraints", "fallback", ) @dataclass(frozen=True) class AgentSpec: # ── Identity ── id: str # slug, globally unique name: str # display name description: str = "" avatar: Optional[str] = None # file path or data URL # ── Prompt(6 段)── prompt_sections: dict[str, str] = field(default_factory=dict) # ── Capability(硬约束)── allowed_toolkits: tuple[str, ...] = ("*",) allowed_delegates: tuple[str, ...] = ("*",) max_iterations: Optional[int] = None default_model: Optional[str] = None # ── Meta ── is_builtin: bool = False source: str = "user_json" # "python" | "user_json" tags: tuple[str, ...] = () created_at: Optional[str] = None updated_at: Optional[str] = None

PYTHON SPEC · 内置 agent 在 prompts/agents/developer.py 里的样子

# 原文件不动 DEVELOPER_PROMPT_SECTIONS,额外导出一个 SPEC: from ..agent_spec import AgentSpec DEVELOPER_PROMPT_SECTIONS = { ... } # 原有 DEVELOPER_SPEC = AgentSpec( id="pupu_developer", name="PuPu Developer", description="全栈编码专家,支持 subagent delegation", prompt_sections=DEVELOPER_PROMPT_SECTIONS, allowed_toolkits=("*",), allowed_delegates=("*",), is_builtin=True, source="python", tags=("coding", "builtin"), ) # analyzer.py: ANALYZER_SPEC = AgentSpec( id="pupu_analyzer", name="PuPu Analyzer", description="只读代码分析专家", prompt_sections=ANALYZER_PROMPT_SECTIONS, allowed_toolkits=("workspace_read_only",), # 见字段说明 allowed_delegates=(), # 分析器不派生 is_builtin=True, source="python", tags=("analysis", "builtin"), )

字段语义说明

字段 语义
id slug,^[a-z][a-z0-9_]{1,39}$。内置 必须以 pupu_ 开头;用户 禁止使用 pupu_ 前缀。Registry 注册冲突时,内置胜出 + 记 warning。
prompt_sections dict,键必须是 6 段之一。缺失的段在组装时跳过,identity 是必填项,其他可选。
allowed_toolkits ("*",) = 任意 toolkit 都允许;() = 禁用所有 toolkit;("workspace",) = 只允许 workspace_toolkit。运行时与 会话选择做交集(硬约束方向:agent 上限卡住会话选择)。
allowed_delegates ("*",) = 可 delegate 到任意池内 agent;() = 不能 delegate;("pupu_analyzer",) = 白名单。运行时组装 SubagentModule 时按这个字段筛选 registry。
max_iterations None = 用 payload.max_iterations(全局默认),非 None = 覆盖。用于给 analyzer 这种设小值。
default_model None = 用 payload.model;非 None = 覆盖(例如某个 agent 坚持用 haiku)。仅 primary 生效,subagent 沿用父 agent 的模型。
is_builtin 前端只读标志 + 防误删。内置=true,用户自建=false。
source "python"(来自 prompts/agents/*.py)或 "user_json"(来自 agent_store)。前端区分 icon,后端选择持久化路径。
avatar 可选。绝对路径 / data URL / null。内置 agent 暂时无头像,前端用 tags 配默认 icon。复用 character_avatars 还是新建 agent_avatars 在 Section 3 定。
